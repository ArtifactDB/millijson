248template<class Input>
249std::string extract_string(Input& input) {
250 size_t start = input.position() + 1;
251 input.advance(); // get past the opening quote.
252 std::string output;
253
254 while (1) {
255 char next = input.get();
256 switch (next) {
257 case '"':
258 input.advance(); // get past the closing quote.
259 return output;
260 case '\\':
261 input.advance();
262 if (!input.valid()) {
263 throw std::runtime_error("unterminated string at position " + std::to_string(start));
264 } else {
265 char next2 = input.get();
266 switch (next2) {
267 case '"':
268 output += '"';
269 break;
270 case 'n':
271 output += '\n';
272 break;
273 case 'r':
274 output += '\r';
275 break;
276 case '\\':
277 output += '\\';
278 break;
279 case '/':
280 output += '/';
281 break;
282 case 'b':
283 output += '\b';
284 break;
285 case 'f':
286 output += '\f';
287 break;
288 case 't':
289 output += '\t';
290 break;
291 case 'u':
292 {
293 unsigned short mb = 0;
294 for (size_t i = 0; i < 4; ++i) {
295 input.advance();
296 if (!input.valid()){
297 throw std::runtime_error("unterminated string at position " + std::to_string(start));
298 }
299 mb *= 16;
300 char val = input.get();
301 switch (val) {
302 case '0': case '1': case '2': case '3': case '4': case '5': case '6': case '7': case '8': case '9':
303 mb += val - '0';
304 break;
305 case 'a': case 'b': case 'c': case 'd': case 'e': case 'f':
306 mb += (val - 'a') + 10;
307 break;
308 case 'A': case 'B': case 'C': case 'D': case 'E': case 'F':
309 mb += (val - 'A') + 10;
310 break;
311 default:
312 throw std::runtime_error("invalid unicode escape detected at position " + std::to_string(input.position() + 1));
313 }
314 }
315
316 // Converting manually from UTF-16 to UTF-8. We only allow
317 // 3 bytes at most because there's only 4 hex digits in JSON.
318 if (mb <= 127) {
319 output += static_cast<char>(mb);
320 } else if (mb <= 2047) {
321 unsigned char left = (mb >> 6) | 0b11000000;
322 output += *(reinterpret_cast<char*>(&left));
323 unsigned char right = (mb & 0b00111111) | 0b10000000;
324 output += *(reinterpret_cast<char*>(&right));
325 } else {
326 unsigned char left = (mb >> 12) | 0b11100000;
327 output += *(reinterpret_cast<char*>(&left));
328 unsigned char middle = ((mb >> 6) & 0b00111111) | 0b10000000;
329 output += *(reinterpret_cast<char*>(&middle));
330 unsigned char right = (mb & 0b00111111) | 0b10000000;
331 output += *(reinterpret_cast<char*>(&right));
332 }
333 }
334 break;
335 default:
336 throw std::runtime_error("unrecognized escape '\\" + std::string(1, next2) + "'");
337 }
338 }
339 break;
340 default:
341 output += next;
342 break;
343 }
344
345 input.advance();
346 if (!input.valid()) {
347 throw std::runtime_error("unterminated string at position " + std::to_string(start));
348 }
349 }
350
351 return output; // Technically unreachable, but whatever.
352}

354template<class Input>
355double extract_number(Input& input) {
356 size_t start = input.position() + 1;
357 double value = 0;
358 double fractional = 10;
359 double exponent = 0;
360 bool negative_exponent = false;
361
362 auto is_terminator = [](char v) -> bool {
363 return v == ',' || v == ']' || v == '}' || isspace(v);
364 };
365
366 bool in_fraction = false;
367 bool in_exponent = false;
368
369 // We assume we're starting from the absolute value, after removing any preceding negative sign.
370 char lead = input.get();
371 if (lead == '0') {
372 input.advance();
373 if (!input.valid()) {
374 return 0;
375 }
376
377 char val = input.get();
378 if (val == '.') {
379 in_fraction = true;
380 } else if (val == 'e' || val == 'E') {
381 in_exponent = true;
382 } else if (is_terminator(val)) {
383 return value;
384 } else {
385 throw std::runtime_error("invalid number starting with 0 at position " + std::to_string(start));
386 }
387
388 } else if (std::isdigit(lead)) {
389 value += lead - '0';
390 input.advance();
391
392 while (input.valid()) {
393 char val = input.get();
394 if (val == '.') {
395 in_fraction = true;
396 break;
397 } else if (val == 'e' || val == 'E') {
398 in_exponent = true;
399 break;
400 } else if (is_terminator(val)) {
401 return value;
402 } else if (!std::isdigit(val)) {
403 throw std::runtime_error("invalid number containing '" + std::string(1, val) + "' at position " + std::to_string(start));
404 }
405 value *= 10;
406 value += val - '0';
407 input.advance();
408 }
409
410 } else {
411 // this should never happen, as extract_number is only called when the lead is a digit (or '-').
412 }
413
414 if (in_fraction) {
415 input.advance();
416 if (!input.valid()) {
417 throw std::runtime_error("invalid number with trailing '.' at position " + std::to_string(start));
418 }
419
420 char val = input.get();
421 if (!std::isdigit(val)) {
422 throw std::runtime_error("'.' must be followed by at least one digit at position " + std::to_string(start));
423 }
424 value += (val - '0') / fractional;
425
426 input.advance();
427 while (input.valid()) {
428 char val = input.get();
429 if (val == 'e' || val == 'E') {
430 in_exponent = true;
431 break;
432 } else if (is_terminator(val)) {
433 return value;
434 } else if (!std::isdigit(val)) {
435 throw std::runtime_error("invalid number containing '" + std::string(1, val) + "' at position " + std::to_string(start));
436 }
437 fractional *= 10;
438 value += (val - '0') / fractional;
439 input.advance();
440 }
441 }
442
443 if (in_exponent) {
444 input.advance();
445 if (!input.valid()) {
446 throw std::runtime_error("invalid number with trailing 'e/E' at position " + std::to_string(start));
447 }
448
449 char val = input.get();
450 if (!std::isdigit(val)) {
451 if (val == '-') {
452 negative_exponent = true;
453 } else if (val != '+') {
454 throw std::runtime_error("'e/E' should be followed by a sign or digit in number at position " + std::to_string(start));
455 }
456 input.advance();
457
458 if (!input.valid()) {
459 throw std::runtime_error("invalid number with trailing exponent sign at position " + std::to_string(start));
460 }
461 val = input.get();
462 if (!std::isdigit(val)) {
463 throw std::runtime_error("exponent sign must be followed by at least one digit in number at position " + std::to_string(start));
464 }
465 }
466
467 exponent += (val - '0');
468 input.advance();
469 while (input.valid()) {
470 char val = input.get();
471 if (is_terminator(val)) {
472 break;
473 } else if (!std::isdigit(val)) {
474 throw std::runtime_error("invalid number containing '" + std::string(1, val) + "' at position " + std::to_string(start));
475 }
476 exponent *= 10;
477 exponent += (val - '0');
478 input.advance();
479 }
480
481 if (exponent) {
482 if (negative_exponent) {
483 exponent *= -1;
484 }
485 value *= std::pow(10.0, exponent);
486 }
487 }
488
489 return value;
490}

565template<class Provisioner, class Input>
566std::shared_ptr<typename Provisioner::base> parse_thing(Input& input) {
567 std::shared_ptr<typename Provisioner::base> output;
568
569 size_t start = input.position() + 1;
570 const char current = input.get();
571
572 if (current == 't') {
573 if (!is_expected_string(input, "true")) {
574 throw std::runtime_error("expected a 'true' string at position " + std::to_string(start));
575 }
576 output.reset(Provisioner::new_boolean(true));
577
578 } else if (current == 'f') {
579 if (!is_expected_string(input, "false")) {
580 throw std::runtime_error("expected a 'false' string at position " + std::to_string(start));
581 }
582 output.reset(Provisioner::new_boolean(false));
583
584 } else if (current == 'n') {
585 if (!is_expected_string(input, "null")) {
586 throw std::runtime_error("expected a 'null' string at position " + std::to_string(start));
587 }
588 output.reset(Provisioner::new_nothing());
589
590 } else if (current == '"') {
591 output.reset(Provisioner::new_string(extract_string(input)));
592
593 } else if (current == '[') {
594 auto ptr = Provisioner::new_array();
595 output.reset(ptr);
596
597 input.advance();
598 chomp(input);
599 if (!input.valid()) {
600 throw std::runtime_error("unterminated array starting at position " + std::to_string(start));
601 }
602
603 if (input.get() != ']') {
604 while (1) {
605 ptr->add(parse_thing<Provisioner>(input));
606
607 chomp(input);
608 if (!input.valid()) {
609 throw std::runtime_error("unterminated array starting at position " + std::to_string(start));
610 }
611
612 char next = input.get();
613 if (next == ']') {
614 break;
615 } else if (next != ',') {
616 throw std::runtime_error("unknown character '" + std::string(1, next) + "' in array at position " + std::to_string(input.position() + 1));
617 }
618
619 input.advance();
620 chomp(input);
621 if (!input.valid()) {
622 throw std::runtime_error("unterminated array starting at position " + std::to_string(start));
623 }
624 }
625 }
626
627 input.advance(); // skip the closing bracket.
628
629 } else if (current == '{') {
630 auto ptr = Provisioner::new_object();
631 output.reset(ptr);
632
633 input.advance();
634 chomp(input);
635 if (!input.valid()) {
636 throw std::runtime_error("unterminated object starting at position " + std::to_string(start));
637 }
638
639 if (input.get() != '}') {
640 while (1) {
641 char next = input.get();
642 if (next != '"') {
643 throw std::runtime_error("expected a string as the object key at position " + std::to_string(input.position() + 1));
644 }
645 auto key = extract_string(input);
646 if (ptr->has(key)) {
647 throw std::runtime_error("detected duplicate keys in the object at position " + std::to_string(input.position() + 1));
648 }
649
650 chomp(input);
651 if (!input.valid()) {
652 throw std::runtime_error("unterminated object starting at position " + std::to_string(start));
653 }
654 if (input.get() != ':') {
655 throw std::runtime_error("expected ':' to separate keys and values at position " + std::to_string(input.position() + 1));
656 }
657
658 input.advance();
659 chomp(input);
660 if (!input.valid()) {
661 throw std::runtime_error("unterminated object starting at position " + std::to_string(start));
662 }
663 ptr->add(std::move(key), parse_thing<Provisioner>(input)); // consuming the key here.
664
665 chomp(input);
666 if (!input.valid()) {
667 throw std::runtime_error("unterminated object starting at position " + std::to_string(start));
668 }
669
670 next = input.get();
671 if (next == '}') {
672 break;
673 } else if (next != ',') {
674 throw std::runtime_error("unknown character '" + std::string(1, next) + "' in array at position " + std::to_string(input.position() + 1));
675 }
676
677 input.advance();
678 chomp(input);
679 if (!input.valid()) {
680 throw std::runtime_error("unterminated object starting at position " + std::to_string(start));
681 }
682 }
683 }
684
685 input.advance(); // skip the closing brace.
686
687 } else if (current == '-') {
688 input.advance();
689 if (!input.valid()) {
690 throw std::runtime_error("incomplete number starting at position " + std::to_string(start));
691 }
692 output.reset(Provisioner::new_number(-extract_number(input)));
693
694 } else if (std::isdigit(current)) {
695 output.reset(Provisioner::new_number(extract_number(input)));
696
697 } else {
698 throw std::runtime_error(std::string("unknown type starting with '") + std::string(1, current) + "' at position " + std::to_string(start));
699 }
700
701 return output;
702}
