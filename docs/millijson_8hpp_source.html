216template<class Input>
217std::string extract_string(Input& input) {
218 size_t start = input.position() + 1;
219 input.advance(); // get past the opening quote.
220 std::string output;
221
222 while (1) {
223 char next = input.get();
224 switch (next) {
225 case '"':
226 input.advance(); // get past the closing quote.
227 return output;
228 case '\\':
229 input.advance();
230 if (!input.valid()) {
231 throw std::runtime_error("unterminated string at position " + std::to_string(start));
232 } else {
233 char next2 = input.get();
234 switch (next2) {
235 case '"':
236 output += '"';
237 break;
238 case 'n':
239 output += '\n';
240 break;
241 case 'r':
242 output += '\r';
243 break;
244 case '\\':
245 output += '\\';
246 break;
247 case '/':
248 output += '/';
249 break;
250 case 'b':
251 output += '\b';
252 break;
253 case 'f':
254 output += '\f';
255 break;
256 case 't':
257 output += '\t';
258 break;
259 // TODO: \u unicode?
260 default:
261 throw std::runtime_error("unrecognized escape '\\" + std::string(1, next2) + "'");
262 }
263 }
264 break;
265 default:
266 output += next;
267 break;
268 }
269
270 input.advance();
271 if (!input.valid()) {
272 throw std::runtime_error("unterminated string at position " + std::to_string(start));
273 }
274 }
275
276 return output; // Technically unreachable, but whatever.
277}

279template<class Input>
280double extract_number(Input& input) {
281 size_t start = input.position() + 1;
282 double value = 0;
283 double fractional = 10;
284 double exponent = 0;
285 bool negative_exponent = false;
286
287 auto is_terminator = [](char v) -> bool {
288 return v == ',' || v == ']' || v == '}' || std::isspace(v);
289 };
290
291 auto finalizer = [&]() -> double {
292 if (exponent) {
293 if (negative_exponent) {
294 exponent *= -1;
295 }
296 return value * std::pow(10.0, exponent);
297 } else {
298 return value;
299 }
300 };
301
302 bool in_fraction = false;
303 bool in_exponent = false;
304
305 // We assume we're starting from the absolute value, after removing any preceding negative sign.
306 char lead = input.get();
307 if (lead == '0') {
308 input.advance();
309 if (!input.valid()) {
310 return 0;
311 }
312
313 char val = input.get();
314 if (val == '.') {
315 in_fraction = true;
316 } else if (val == 'e' || val == 'E') {
317 in_exponent = true;
318 } else if (is_terminator(val)) {
319 return finalizer();
320 } else {
321 throw std::runtime_error("invalid number starting with 0 at position " + std::to_string(start));
322 }
323
324 } else if (std::isdigit(lead)) {
325 value += lead - '0';
326 input.advance();
327
328 while (input.valid()) {
329 char val = input.get();
330 if (val == '.') {
331 in_fraction = true;
332 break;
333 } else if (val == 'e' || val == 'E') {
334 in_exponent = true;
335 break;
336 } else if (is_terminator(val)) {
337 return finalizer();
338 } else if (!std::isdigit(val)) {
339 throw std::runtime_error("invalid number containing '" + std::string(1, val) + "' at position " + std::to_string(start));
340 }
341 value *= 10;
342 value += val - '0';
343 input.advance();
344 }
345
346 } else {
347 // this should never happen, as extract_number is only called when the lead is a digit (or '-').
348 }
349
350 if (in_fraction) {
351 input.advance();
352 if (!input.valid()) {
353 throw std::runtime_error("invalid number with trailing '.' at position " + std::to_string(start));
354 }
355
356 char val = input.get();
357 if (!std::isdigit(val)) {
358 throw std::runtime_error("'.' must be followed by at least one digit at position " + std::to_string(start));
359 }
360 value += (val - '0') / fractional;
361
362 input.advance();
363 while (input.valid()) {
364 char val = input.get();
365 if (val == 'e' || val == 'E') {
366 in_exponent = true;
367 break;
368 } else if (is_terminator(val)) {
369 return finalizer();
370 } else if (!std::isdigit(val)) {
371 throw std::runtime_error("invalid number containing '" + std::string(1, val) + "' at position " + std::to_string(start));
372 }
373 fractional *= 10;
374 value += (val - '0') / fractional;
375 input.advance();
376 }
377 }
378
379 if (in_exponent) {
380 input.advance();
381 if (!input.valid()) {
382 throw std::runtime_error("invalid number with trailing 'e/E' at position " + std::to_string(start));
383 }
384
385 char val = input.get();
386 if (!std::isdigit(val)) {
387 if (val == '-') {
388 negative_exponent = true;
389 } else if (val != '+') {
390 throw std::runtime_error("'e/E' must be followed by a sign in number at position " + std::to_string(start));
391 }
392 input.advance();
393
394 if (!input.valid()) {
395 throw std::runtime_error("invalid number with trailing exponent sign at position " + std::to_string(start));
396 }
397 val = input.get();
398 if (!std::isdigit(val)) {
399 throw std::runtime_error("exponent sign must be followed by at least one digit in number at position " + std::to_string(start));
400 }
401 }
402
403 exponent += (val - '0');
404 input.advance();
405 while (input.valid()) {
406 char val = input.get();
407 if (is_terminator(val)) {
408 return finalizer();
409 } else if (!std::isdigit(val)) {
410 throw std::runtime_error("invalid number containing '" + std::string(1, val) + "' at position " + std::to_string(start));
411 }
412 exponent *= 10;
413 exponent += (val - '0');
414 input.advance();
415 }
416 }
417
418 return finalizer();
419}

422std::shared_ptr<Base> parse_thing(Input& input) {
423 std::shared_ptr<Base> output;
424
425 size_t start = input.position() + 1;
426 const char current = input.get();
427
428 if (current == 't') {
429 if (!is_expected_string(input, "true")) {
430 throw std::runtime_error("expected a 'true' string at position " + std::to_string(start));
431 }
432 output.reset(new Boolean(true));
433
434 } else if (current == 'f') {
435 if (!is_expected_string(input, "false")) {
436 throw std::runtime_error("expected a 'false' string at position " + std::to_string(start));
437 }
438 output.reset(new Boolean(false));
439
440 } else if (current == 'n') {
441 if (!is_expected_string(input, "null")) {
442 throw std::runtime_error("expected a 'null' string at position " + std::to_string(start));
443 }
444 output.reset(new Nothing);
445
446 } else if (current == '"') {
447 output.reset(new String(extract_string(input)));

454 chomp(input);
455 if (!input.valid()) {

457 }
458
459 if (input.get() != ']') {

461 ptr->values.push_back(parse_thing(input));
462

464 if (!input.valid()) {

466 }
467

469 if (next == ']') {
470 break;
471 } else if (next != ',') {

473 }
474

476 chomp(input);

479 }

485 } else if (current == '{') {

519 }

521
522 chomp(input);
523 if (!input.valid()) {
524 throw std::runtime_error("unterminated object starting at position " + std::to_string(start));
525 }
526
527 next = input.get();
528 if (next == '}') {
529 break;
530 } else if (next != ',') {
531 throw std::runtime_error("unknown character '" + std::string(1, next) + "' in array at position " + std::to_string(input.position() + 1));
532 }
533
534 input.advance();
535 chomp(input);

537 throw std::runtime_error("unterminated object starting at position " + std::to_string(start));
538 }
539 }
540 }

543
544 } else if (current == '-') {
545 input.advance();
546 if (!input.valid()) {
547 throw std::runtime_error("incomplete number starting at position " + std::to_string(start));
548 }
549 output.reset(new Number(-extract_number(input)));

551 } else if (std::isdigit(current)) {
552 output.reset(new Number(extract_number(input)));

554 } else {
