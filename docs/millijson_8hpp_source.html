216// Return value of the various chomp functions indicates whether there are any
217// characters left in 'input', allowing us to avoid an extra call to valid().
218template<class Input_>
219bool raw_chomp(Input_& input, bool ok) {
220 while (ok) {
221 switch(input.get()) {
222 // Allowable whitespaces as of https://www.rfc-editor.org/rfc/rfc7159#section-2.
223 case ' ': case '\n': case '\r': case '\t':
224 break;
225 default:
226 return true;
227 }
228 ok = input.advance();
229 }
230 return false;
231}

249template<class Input_>
250bool is_expected_string(Input_& input, const char* ptr, size_t len) {
251 // We use a hard-coded 'len' instead of scanning for '\0' to enable loop unrolling.
252 for (size_t i = 1; i < len; ++i) {
253 // The current character was already used to determine what string to
254 // expect, so we can skip past it in order to match the rest of the
255 // string. This is also why we start from i = 1 instead of i = 0.
256 if (!input.advance()) {
257 return false;
258 }
259 if (input.get() != ptr[i]) {
260 return false;
261 }
262 }
263 input.advance(); // move off the last character.
264 return true;
265}

586template<class Provisioner_, class Input_>
587std::shared_ptr<typename Provisioner_::Base> parse_thing(Input_& input) {
588 std::shared_ptr<typename Provisioner_::Base> output;
589
590 size_t start = input.position() + 1;
591 const char current = input.get();
592
593 switch(current) {
594 case 't':
595 if (!is_expected_string(input, "true", 4)) {
596 throw std::runtime_error("expected a 'true' string at position " + std::to_string(start));
597 }
598 output.reset(Provisioner_::new_boolean(true));
599 break;
600
601 case 'f':
602 if (!is_expected_string(input, "false", 5)) {
603 throw std::runtime_error("expected a 'false' string at position " + std::to_string(start));
604 }
605 output.reset(Provisioner_::new_boolean(false));
606 break;
607
608 case 'n':
609 if (!is_expected_string(input, "null", 4)) {
610 throw std::runtime_error("expected a 'null' string at position " + std::to_string(start));
611 }
612 output.reset(Provisioner_::new_nothing());
613 break;
614
615 case '"':
616 output.reset(Provisioner_::new_string(extract_string(input)));
617 break;
618
619 case '[':
620 {
621 std::vector<std::shared_ptr<typename Provisioner_::Base> > contents;
622
623 if (!advance_and_chomp(input)) {
624 throw std::runtime_error("unterminated array starting at position " + std::to_string(start));
625 }
626
627 if (input.get() != ']') {
628 while (1) {
629 contents.push_back(parse_thing<Provisioner_>(input));
630
631 if (!check_and_chomp(input)) {
632 throw std::runtime_error("unterminated array starting at position " + std::to_string(start));
633 }
634
635 char next = input.get();
636 if (next == ']') {
637 break;
638 } else if (next != ',') {
639 throw std::runtime_error("unknown character '" + std::string(1, next) + "' in array at position " + std::to_string(input.position() + 1));
640 }
641
642 if (!advance_and_chomp(input)) {
643 throw std::runtime_error("unterminated array starting at position " + std::to_string(start));
644 }
645 }
646 }
647
648 output.reset(Provisioner_::new_array(std::move(contents)));
649 }
650 input.advance(); // skip the closing bracket.
651 break;
652
653 case '{':
654 {
655 std::unordered_map<std::string, std::shared_ptr<typename Provisioner_::Base> > contents;
656
657 if (!advance_and_chomp(input)) {
658 throw std::runtime_error("unterminated object starting at position " + std::to_string(start));
659 }
660
661 if (input.get() != '}') {
662 while (1) {
663 char next = input.get();
664 if (next != '"') {
665 throw std::runtime_error("expected a string as the object key at position " + std::to_string(input.position() + 1));
666 }
667 auto key = extract_string(input);
668 if (contents.find(key) != contents.end()) {
669 throw std::runtime_error("detected duplicate keys in the object at position " + std::to_string(input.position() + 1));
670 }
671
672 if (!check_and_chomp(input)) {
673 throw std::runtime_error("unterminated object starting at position " + std::to_string(start));
674 }
675 if (input.get() != ':') {
676 throw std::runtime_error("expected ':' to separate keys and values at position " + std::to_string(input.position() + 1));
677 }
678
679 if (!advance_and_chomp(input)) {
680 throw std::runtime_error("unterminated object starting at position " + std::to_string(start));
681 }
682 contents[std::move(key)] = parse_thing<Provisioner_>(input); // consuming the key here.
683
684 if (!check_and_chomp(input)) {
685 throw std::runtime_error("unterminated object starting at position " + std::to_string(start));
686 }
687
688 next = input.get();
689 if (next == '}') {
690 break;
691 } else if (next != ',') {
692 throw std::runtime_error("unknown character '" + std::string(1, next) + "' in array at position " + std::to_string(input.position() + 1));
693 }
694
695 if (!advance_and_chomp(input)) {
696 throw std::runtime_error("unterminated object starting at position " + std::to_string(start));
697 }
698 }
699 }
700
701 output.reset(Provisioner_::new_object(std::move(contents)));
702 }
703 input.advance(); // skip the closing brace.
704 break;
705
706 case '-':
707 if (!input.advance()) {
708 throw std::runtime_error("incomplete number starting at position " + std::to_string(start));
709 }
710 if (!is_digit(input.get())) {
711 throw std::runtime_error("invalid number starting at position " + std::to_string(start));
712 }
713 output.reset(Provisioner_::new_number(-extract_number(input)));
714 break;
715
716 case '0': case '1': case '2': case '3': case '4': case '5': case '6': case '7': case '8': case '9':
717 output.reset(Provisioner_::new_number(extract_number(input)));
718 break;
719
720 default:
721 throw std::runtime_error(std::string("unknown type starting with '") + std::string(1, current) + "' at position " + std::to_string(start));
722 }
723
724 return output;
725}

896public:
897 FileReader(const char* path, size_t buffer_size) : my_handle(std::fopen(path, "rb")), my_buffer(buffer_size) {
898 if (!my_handle) {
899 throw std::runtime_error("failed to open file at '" + std::string(path) + "'");
900 }
901 fill();
902 }

904 ~FileReader() {

908private:
909 FILE* my_handle;
910 std::vector<char> my_buffer;
911 size_t my_available = 0;
912 size_t my_index = 0;
913 size_t my_overall = 0;
914 bool my_finished = false;
915
916public:

918 return my_buffer[my_index];
919 }
920
921 bool valid() const {
922 return my_index < my_available;
923 }
924
925 bool advance() {
926 ++my_index;
927 if (my_index < my_available) {
928 return true;
929 }
930

932 my_overall += my_available;
933 fill();
934 return valid();

937 void fill() {
938 if (my_finished) {
939 my_available = 0;
940 return;
941 }
942
943 my_available = std::fread(my_buffer.data(), sizeof(char), my_buffer.size(), my_handle);
944 if (my_available == my_buffer.size()) {
945 return;
946 }
